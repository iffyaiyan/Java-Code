/*1. You are given a number n.
2. You've to create a pattern as shown in output format*/




import java.util.*;

class GFG {
    
    
    
	public static void main (String[] args) {
    
        Scanner scn = new Scanner(System.in);
        
        int x = scn.nextInt();
        
        for (int i = 1; i<=10; i++)
        {
            int v = x*i;
            System.out.println(x + " * " + i + " = " + v);
        }
        
    }
    
}
